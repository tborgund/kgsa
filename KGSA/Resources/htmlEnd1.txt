$('.behandlet').click( function() {  

	window.external.callBehandlet( $(this).attr('data'), $(this).text() );
	if ($(this).text() == 'Ja') {
		$(this).text('Nei');
		$(this).parents("tr").removeClass("behandletService");
	}
	else {
		$(this).text('Ja');
		$(this).parents("tr").addClass("behandletService");
	}
	return false;
});
